    function go(id) {
        document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
        document.getElementById(id).classList.add('active');
        window.scrollTo(0,0);
    }

    function filterDrugs() {
        const input = document.getElementById('search-input').value.toLowerCase();
        const buttons = document.querySelectorAll('.menu-btn');
        buttons.forEach(btn => {
            const name = btn.getAttribute('data-name');
            btn.style.display = name.includes(input) ? "flex" : "none";
        });
    }

    function calcAll(val) {
        if(val !== undefined) document.querySelectorAll('.peso').forEach(i => i.value = val);
        const peso = parseFloat(document.querySelector('.peso').value);
        if(!peso || peso <= 0) return;

        const g = (id) => {
            const el = document.getElementById(id);
            return el ? parseFloat(el.value) : 0;
        };
        const s = (id, txt) => { const el = document.getElementById(id); if(el) el.innerText = txt; };

        // mcg/kg/min
        ['nora','dobuta','dopa','nitroP','remi','milri','feni','esmo','alfe'].forEach(d => {
            const elM = document.getElementById(d+'_m');
            if(elM) s('res_'+d, ((g(d+'_d') * peso * 60) / ((g(d+'_m') * 1000) / g(d+'_v'))).toFixed(1));
        });
        
        // mg/kg/h ou mcg/kg/h
        ['mida','fenta','dex','morfi'].forEach(d => {
            const elM = document.getElementById(d+'_m');
            if(elM) s('res_'+d, ((g(d+'_d') * peso) / (g(d+'_m') / g(d+'_v'))).toFixed(1));
        });

        // Especificos
        const nG = document.getElementById('nitroG_m');
        if(nG) s('res_nitroG', ((g('nitroG_d') * 60) / ((g('nitroG_m') * 1000) / g('nitroG_v'))).toFixed(1));
        
        const pC = document.getElementById('prop_c');
        if(pC) {
            let pc = g('prop_c'); 
            s('res_prop_i', ((g('prop_i')*peso)/pc).toFixed(1)); 
            s('res_prop_v', ((g('prop_m')*peso*60)/(pc*1000)).toFixed(1));
        }

        const rC = document.getElementById('roc_c');
        if(rC) {
            let rc = g('roc_c'); 
            s('res_roc_b', ((g('roc_a')*peso)/rc).toFixed(1)); 
            s('res_roc_v', ((g('roc_m')*peso)/rc).toFixed(1));
        }
        
        if(document.getElementById('cisa_c')){
            let cc = g('cisa_c');
            s('res_cisa_b', ((g('cisa_a')*peso)/cc).toFixed(1));
            s('res_cisa_v', ((g('cisa_d')*peso)/cc).toFixed(1));
        }
    }
</script>
